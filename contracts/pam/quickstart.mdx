---
title: "Quickstart"
description: "End-to-end example: create a resolution, place a bet, submit the outcome, and settle."
---

This example walks through the full PAM lifecycle — from creating a resolution to settling a bet.

```typescript
import * as anchor from "@coral-xyz/anchor";
import { Program } from "@coral-xyz/anchor";
import type { SoftsolPam } from "./types/softsol_pam";
import idl from "./idl/softsol_pam.json";

const program = new Program<SoftsolPam>(idl as SoftsolPam, { connection });

// ── Load keypairs ─────────────────────────────────────────────────────────────
// platform  — the wallet that owns and pays for this resolution
// authority — the resolver that will later submit the outcome (can equal platform)
// user      — the bettor
// settler   — anyone can settle once the resolution is finalized

const platform  = anchor.web3.Keypair.fromSecretKey(Uint8Array.from(/* your platform keypair */));
const authority = anchor.web3.Keypair.fromSecretKey(Uint8Array.from(/* your authority keypair */));
const user      = anchor.web3.Keypair.fromSecretKey(Uint8Array.from(/* your user keypair */));
const settler   = anchor.web3.Keypair.fromSecretKey(Uint8Array.from(/* your settler keypair */));

const OPERATOR = new anchor.web3.PublicKey("cxa3sYKNMk5L8PvDqEzWWLAfd4hr88d6U25W6Yz3FiE");
const id = new anchor.BN(1);

// ── Step 1: Create the resolution ────────────────────────────────────────────
// resolution PDA is auto-derived by Anchor from [ss_resolution, platform, id]

await program.methods
  .requestResolution(
    id,
    { singleSource: { authority: authority.publicKey, deadline: null, coAuthority: null } },
    Buffer.from("q: Will team A beat team B?\no: yes, no", "utf-8")
  )
  .accounts({ platform: platform.publicKey })
  .signers([platform])
  .rpc();

// ── Step 2: Fetch the resolution address for subsequent calls ─────────────────

const [resolution] = anchor.web3.PublicKey.findProgramAddressSync(
  [Buffer.from("ss_resolution"), platform.publicKey.toBuffer(), id.toArrayLike(Buffer, "le", 8)],
  program.programId
);

// ── Step 3: Place a bet ───────────────────────────────────────────────────────
// Bet 10 USDC on slot 0 == 1_000_000 (yes wins), pays out 25 USDC

await program.methods
  .place(
    new anchor.BN(10_000_000),
    id,
    [
      {
        conditions: [{ o1: 0, o2: null, op: { eq: {} }, value: new anchor.BN(1_000_000) }],
        payoutAmount: new anchor.BN(25_000_000),
      },
    ]
  )
  .accounts({ operator: OPERATOR, platform: platform.publicKey, user: user.publicKey, resolution })
  .signers([user])
  .rpc();

// ── Step 4: Submit the outcome ────────────────────────────────────────────────
// resolved_slots[0] = 1_000_000 → "yes"

await program.methods
  .submitResolution([new anchor.BN(1_000_000)], null, null, null)
  .accounts({ authority: authority.publicKey, resolution })
  .signers([authority])
  .rpc();

// ── Step 5: Settle ────────────────────────────────────────────────────────────
// slot 0 == 1_000_000 → row matches → user receives 25 USDC

await program.methods
  .settle(id)
  .accounts({ settler: settler.publicKey, platform: platform.publicKey, user: user.publicKey, resolution })
  .signers([settler])
  .rpc();
```
